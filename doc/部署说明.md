# 家庭药品管理系统 - 部署说明

## 一、环境要求

### 1.1 后端环境
- **JDK**: 1.8
- **MySQL**: 5.7
- **Redis**: 6.0+（可选，用于验证码缓存）
- **Maven**: 3.8+

### 1.2 前端环境
- **Node.js**: 16+
- **npm**: 8+ 或 **yarn**: 1.22+

---

## 二、数据库初始化

### 2.1 创建数据库
```sql
CREATE DATABASE medi_home CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
```

### 2.2 执行初始化脚本
```sql
USE medi_home;

-- 用户表
CREATE TABLE sys_user (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    phone VARCHAR(20) NOT NULL UNIQUE COMMENT '手机号',
    password VARCHAR(100) NOT NULL COMMENT '密码',
    nickname VARCHAR(50) COMMENT '昵称',
    avatar VARCHAR(255) COMMENT '头像',
    role TINYINT DEFAULT 1 COMMENT '角色：0-管理员，1-普通用户',
    status TINYINT DEFAULT 1 COMMENT '状态：0-禁用，1-正常',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    update_time DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_phone (phone)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

-- 家庭成员表
CREATE TABLE family_member (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL COMMENT '创建者用户ID',
    name VARCHAR(50) NOT NULL COMMENT '成员姓名',
    age INT COMMENT '年龄',
    gender TINYINT COMMENT '性别：0-女，1-男',
    allergy_history VARCHAR(500) COMMENT '过敏史',
    common_drugs VARCHAR(500) COMMENT '常用药品',
    is_default TINYINT DEFAULT 0 COMMENT '是否默认成员：0-否，1-是',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_id (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='家庭成员表';

-- 药品分类表
CREATE TABLE drug_category (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL DEFAULT 0 COMMENT '创建者ID（0为系统默认）',
    name VARCHAR(50) NOT NULL COMMENT '分类名称',
    sort_order INT DEFAULT 0 COMMENT '排序号',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_id (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='药品分类表';

-- 药品表
CREATE TABLE drug (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL COMMENT '创建者ID',
    category_id BIGINT COMMENT '分类ID',
    name VARCHAR(100) NOT NULL COMMENT '药品名称',
    specification VARCHAR(100) COMMENT '规格',
    manufacturer VARCHAR(100) COMMENT '生产厂家',
    produce_date DATE COMMENT '生产日期',
    expiry_date DATE NOT NULL COMMENT '有效期',
    usage_dosage VARCHAR(500) COMMENT '用法用量',
    indication VARCHAR(500) COMMENT '适应症',
    contraindication VARCHAR(500) COMMENT '禁忌',
    storage_condition VARCHAR(100) COMMENT '储存条件',
    purchase_date DATE COMMENT '采购日期',
    stock_quantity INT DEFAULT 0 COMMENT '库存数量',
    stock_unit VARCHAR(20) COMMENT '库存单位',
    location VARCHAR(100) COMMENT '存放位置',
    images VARCHAR(1000) COMMENT '图片URL（JSON数组）',
    status TINYINT DEFAULT 0 COMMENT '状态：0-正常，1-即将过期，2-已过期，3-已归档',
    remark VARCHAR(500) COMMENT '备注',
    is_favorite TINYINT DEFAULT 0 COMMENT '是否收藏：0-否，1-是',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    update_time DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_user_id (user_id),
    INDEX idx_category_id (category_id),
    INDEX idx_status (status),
    INDEX idx_expiry_date (expiry_date)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='药品表';

-- 库存变动记录表
CREATE TABLE drug_stock_log (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    drug_id BIGINT NOT NULL COMMENT '药品ID',
    change_type TINYINT NOT NULL COMMENT '变动类型：1-增加，2-减少，3-盘点调整',
    change_quantity INT NOT NULL COMMENT '变动数量',
    before_quantity INT NOT NULL COMMENT '变动前数量',
    after_quantity INT NOT NULL COMMENT '变动后数量',
    reason VARCHAR(200) COMMENT '变动原因',
    operator_id BIGINT NOT NULL COMMENT '操作人ID',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_drug_id (drug_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='库存变动记录表';

-- 用药记录表
CREATE TABLE medication_record (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL COMMENT '记录创建者ID',
    member_id BIGINT NOT NULL COMMENT '用药成员ID',
    drug_id BIGINT NOT NULL COMMENT '药品ID',
    dosage VARCHAR(50) NOT NULL COMMENT '用药剂量',
    medication_time DATETIME NOT NULL COMMENT '用药时间',
    remark VARCHAR(500) COMMENT '备注',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_id (user_id),
    INDEX idx_member_id (member_id),
    INDEX idx_drug_id (drug_id),
    INDEX idx_medication_time (medication_time)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用药记录表';

-- 用药提醒表
CREATE TABLE medication_reminder (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL COMMENT '创建者ID',
    member_id BIGINT NOT NULL COMMENT '用药成员ID',
    drug_id BIGINT NOT NULL COMMENT '药品ID',
    dosage VARCHAR(50) NOT NULL COMMENT '用药剂量',
    reminder_time TIME NOT NULL COMMENT '提醒时间',
    frequency_type TINYINT DEFAULT 1 COMMENT '频率类型：1-每日，2-每周，3-自定义',
    frequency_value VARCHAR(50) COMMENT '频率值',
    status TINYINT DEFAULT 1 COMMENT '状态：0-暂停，1-正常',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_id (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用药提醒表';

-- 系统消息表
CREATE TABLE sys_message (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL COMMENT '接收用户ID',
    type TINYINT NOT NULL COMMENT '消息类型：1-过期提醒，2-库存预警，3-系统通知',
    title VARCHAR(100) NOT NULL COMMENT '消息标题',
    content VARCHAR(500) NOT NULL COMMENT '消息内容',
    related_id BIGINT COMMENT '关联业务ID',
    is_read TINYINT DEFAULT 0 COMMENT '是否已读：0-否，1-是',
    create_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_id (user_id),
    INDEX idx_is_read (is_read)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统消息表';

-- 系统配置表
CREATE TABLE sys_config (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL DEFAULT 0 COMMENT '用户ID（0为系统默认）',
    config_key VARCHAR(50) NOT NULL COMMENT '配置键',
    config_value VARCHAR(200) NOT NULL COMMENT '配置值',
    remark VARCHAR(200) COMMENT '备注',
    UNIQUE KEY uk_user_key (user_id, config_key)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统配置表';

-- 插入默认分类

```

## 三、后端部署

### 3.1 修改配置文件
编辑 `backend/src/main/resources/application.yml`：
```yaml
spring:
  datasource:
    url: jdbc:mysql://localhost:3306/medi_home?useUnicode=true&characterEncoding=utf-8&useSSL=false&serverTimezone=Asia/Shanghai
    username: your_username      # 修改为你的MySQL用户名
    password: your_password      # 修改为你的MySQL密码
  
  redis:
    host: localhost              # Redis地址（可选）
    port: 6379
    password: 

jwt:
  secret: your-secret-key-here   # 建议修改为复杂的密钥
  expiration: 604800000

upload:
  path: ./uploads                # 文件上传路径
```

### 3.2 编译打包
```bash
cd backend
mvn clean package -DskipTests
```

### 3.3 运行项目
```bash
# 方式1：直接运行
java -jar target/medi-home-1.0.0.jar

# 方式2：指定配置文件运行
java -jar target/medi-home-1.0.0.jar --spring.config.location=application-prod.yml

# 方式3：后台运行（Linux）
nohup java -jar target/medi-home-1.0.0.jar > app.log 2>&1 &
```

### 3.4 验证部署
访问 `http://localhost:8080`，如果看到 401 错误页面，说明服务启动成功。

## 四、前端部署

### 4.1 安装依赖
```bash
cd frontend
npm install
```

### 4.2 修改配置（可选）
编辑 `frontend/vue.config.js`：
```javascript
devServer: {
  port: 3000,
  proxy: {
    '/api': {
      target: 'http://localhost:8080',  // 修改为后端实际地址
      changeOrigin: true
    }
  }
}
```

### 4.3 开发环境运行
```bash
npm run serve
```
访问 `http://localhost:3000`

### 4.4 生产环境构建
```bash
npm run build
```
构建后的文件在 `frontend/dist` 目录，可部署到 Nginx 等 Web 服务器。

### 4.5 Nginx 配置示例
```nginx
server {
    listen 80;
    server_name your-domain.com;
    
    location / {
        root /path/to/frontend/dist;
        index index.html;
        try_files $uri $uri/ /index.html;
    }
    
    location /api {
        proxy_pass http://localhost:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
    
    location /uploads {
        alias /path/to/backend/uploads;
    }
}
```

## 五、Docker 部署（可选）

### 5.1 Dockerfile（后端）
```dockerfile
FROM openjdk:8-jdk-slim
COPY target/medi-home-1.0.0.jar app.jar
EXPOSE 8080
ENTRYPOINT ["java", "-jar", "/app.jar"]
```

### 5.2 docker-compose.yml
```yaml
version: '3.8'
services:
  mysql:
    image: mysql:5.7
    environment:
      MYSQL_ROOT_PASSWORD: root
      MYSQL_DATABASE: medi_home
    ports:
      - "3306:3306"
    volumes:
      - mysql_data:/var/lib/mysql

  redis:
    image: redis:6-alpine
    ports:
      - "6379:6379"

  backend:
    build: ./backend
    ports:
      - "8080:8080"
    depends_on:
      - mysql
      - redis
    environment:
      SPRING_DATASOURCE_URL: jdbc:mysql://mysql:3306/medi_home
      SPRING_REDIS_HOST: redis

  frontend:
    build: ./frontend
    ports:
      - "80:80"
    depends_on:
      - backend

volumes:
  mysql_data:
```
